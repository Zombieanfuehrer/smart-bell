set(LIBRARY_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/blink.cpp")
set(LIBRARY_HEADERS "${PROJECT_SOURCE_DIR}/public/blink.h")
set(LIBRARY_INCLUDES "${PROJECT_SOURCE_DIR}/public" "${CMAKE_BINARY_DIR}/configured/include")

add_library("${LIBRARY_NAME}" STATIC ${LIBRARY_SOURCES} ${LIBRARY_HEADERS})
target_include_directories("${LIBRARY_NAME}" PUBLIC ${LIBRARY_INCLUDES})

install(TARGETS ${LIBRARY_NAME} DESTINATION lib)
install(FILES ${LIBRARY_HEADERS} DESTINATION include)
